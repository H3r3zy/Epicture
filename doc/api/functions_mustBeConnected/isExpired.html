isExpired() {
  var now = DateTime.now();
  var expiredms = globals.createdAt.millisecondsSinceEpoch + globals.expiresIn * 1000;
  var expiredAt = DateTime.fromMillisecondsSinceEpoch(expiredms);

  return (now.compareTo(expiredAt) > 0);
}
