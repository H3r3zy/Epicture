getTagSearch(int page) async {
  this.page = page;
  var images = await Imgur.tagSearch(tag: this.tag, page: this.page);
  images = images["items"];

  if (!this.mounted)
    return;

  setState(() {
    if (data == null)
      data = images;
    else
      data = new List.from(data)..addAll(images);
  });
}
